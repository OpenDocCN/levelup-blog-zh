编辑Bash配置文件:

  1. 从这些指令下面复制代码
  2. 将代码粘贴到文本编辑器中
  3. 点击“保存”
# Pyenv environment variables
export PYENV_ROOT="$HOME/.pyenv"
export PATH="$PYENV_ROOT/bin:$PATH"# Pyenv initialization
if command -v pyenv 1>/dev/null 2>&1; then
  eval "$(pyenv init -)"
fi
