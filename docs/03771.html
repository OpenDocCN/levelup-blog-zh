            std::make_shared<ThreadFuncImpl>([me, cb, login]() {
        try {

                me->threads->run_on_main_thread(
                    std::make_shared<ThreadFuncImpl>([cb, user]() {
                        cb->on_success(user);

                     std::make_shared<ThreadFuncImpl>([cb, user]() {
                        cb->on_success(user.value());
                    }));

            me->threads->run_on_main_thread(
                  std::make_shared<ThreadFuncImpl>([cb]() {
                cb->on_error("could not load user");
            }));
        }
    }));
