解决办法

class Node:
  def __init__(self, val = 0, next=None): 
    self.val = val
    self.next = next
      
def MergeTwoLists(ll1, ll2):
    current = dummy = Node()
    while ll1 and ll2:
        if ll1.val < ll2.val:
            current.next = ll1
            ll1 = ll1.next
        else:
            current.next = ll2
            ll2 = ll2.next
        current = current.next
    
    if ll1 or ll2:
        current.next = ll1 if ll1 else ll2
            
    return dummy.nextLL1 = Node(1, Node(2, Node(4)))
LL2 = Node(1, Node(3, Node(4)))
